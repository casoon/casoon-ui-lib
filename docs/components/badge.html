Badge

Die Badge-Komponente wird verwendet, um kleine Informationsblöcke oder Statusindikatoren anzuzeigen.

<!-- Mit Buttons -->
<button class="btn">
    Benachrichtigungen
    <span class="badge badge-primary">3</span>
</button>

<!-- Mit Icons -->
<span class="icon">🔔</span>
<span class="badge badge-error">5</span>

<!-- Mit Text -->
<p>
    Neuer Artikel
    <span class="badge badge-success">Neu</span>
</p>

<!-- Mit Cards -->
<div class="card">
    <div class="card-header">
        <h3>Projekt Status</h3>
        <span class="badge badge-warning">In Bearbeitung</span>
    </div>
    <div class="card-body">
        <p>Projektbeschreibung...</p>
    </div>
</div>
