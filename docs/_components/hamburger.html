Verwendung mit Alpine.js

Der Hamburger-Button kann mit Alpine.js f端r die Sidebar-Steuerung verwendet werden:

<button 
    class="hamburger hamburger--squeeze" 
    type="button"
    :class="{ 'is-active': isOpen }"
    @click="isOpen = !isOpen">
    <span class="hamburger-box">
        <span class="hamburger-inner"></span>
    </span>
</button>

<!-- Sidebar -->
<div class="sidebar" x-show="isOpen">
    <!-- Sidebar-Inhalt -->
</div>
