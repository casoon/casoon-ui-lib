Benutzerdefinierter Slider

<div class="slider-custom">
    <input type="range" class="slider" min="0" max="100" value="50">
    <div class="slider-value"></div>
</div>

<script>
const slider = document.querySelector('.slider-custom input');
const value = document.querySelector('.slider-value');

slider.addEventListener('input', () => {
    value.textContent = slider.value;
});
</script>

Verwendung mit JavaScript

Event-Handling für Slider:

const slider = document.querySelector('.slider');

// Wert-Änderungen überwachen
slider.addEventListener('input', (e) => {
    console.log('Aktueller Wert:', e.target.value);
});

// Änderung abgeschlossen
slider.addEventListener('change', (e) => {
    console.log('Finaler Wert:', e.target.value);
});
